cmake_minimum_required(VERSION 2.8)
project(g2o_curve_fitting)
set(CMAKE_BUILD_TYPE "Release")
set(CMAKE_CXX_FLAGS "-std=c++ -O3")
list(APPEND CMAKE_MODULE_PATH ${PROJECT_SOURCE_DIR}/cmake_modules)
fing_package(G2O REQUIED)
include_directories(${G2O_INCLUDE_DIRS} "/usr/include/eigen3")
fing_package(OpenCV REQUIED)
include_directories(${OpenCV_DIRS})
add_executable(g2o_boke main.cpp)
target_link_libraries(g2o_boke ${OpenCV_LIBS} g2o_core g2o_stuff)
